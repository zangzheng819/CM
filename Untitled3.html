import numpy as np
import matplotlib.pyplot as plt
from scipy.integrate import odeint

# Given data
m = 0.5  # mass of the grenade in kg
v0 = np.array([30, 30])  # initial velocity in m/s
g = 9.8  # gravitational acceleration in m/s^2
c = 2.5e-3  # air resistance coefficient in N·s^2/m^2
t_explosion = 4.5  # time of explosion in seconds
delta_v = np.array([30, 30])  # velocity change due to explosion

# Differential equations for motion with air resistance
def equations(w, t):
    x, y, vx, vy = w
    speed = np.sqrt(vx**2 + vy**2)
    dvx_dt = - (c / m) * speed * vx
    dvy_dt = - (c / m) * speed * vy - g
    return [vx, vy, dvx_dt, dvy_dt]

# Time points for simulation
t = np.linspace(0, 6, 1000)

# Initial conditions [x0, y0, vx0, vy0]
initial_conditions = [0, 0, v0[0], v0[1]]

# Solve the differential equations
solution = odeint(equations, initial_conditions, t)

# Extracting the positions and velocities
x = solution[:, 0]
y = solution[:, 1]
vx = solution[:, 2]
vy = solution[:, 3]

# Find velocity before the explosion at t = 4.5 s
idx_explosion = np.argmin(np.abs(t - t_explosion))
v_before_explosion = np.sqrt(vx[idx_explosion]**2 + vy[idx_explosion]**2)

# After the explosion, velocities of the two fragments
v_fragment1 = np.array([vx[idx_explosion], vy[idx_explosion]]) + delta_v
v_fragment2 = np.array([vx[idx_explosion], vy[idx_explosion]]) - delta_v  # for momentum conservation

# Function to get the trajectory of each fragment after the explosion
def fragment_trajectory(v_fragment, t_after, initial_pos):
    vx_f, vy_f = v_fragment
    x0_f, y0_f = initial_pos
    x_f = x0_f + vx_f * t_after
    y_f = y0_f + vy_f * t_after - 0.5 * g * t_after**2
    return x_f, y_f

# Time after the explosion
t_after = t[t > t_explosion] - t_explosion

# Initial positions at the explosion point
x_explosion, y_explosion = x[idx_explosion], y[idx_explosion]

# Get trajectories of each fragment after the explosion
x_f1, y_f1 = fragment_trajectory(v_fragment1, t_after, [x_explosion, y_explosion])
x_f2, y_f2 = fragment_trajectory(v_fragment2, t_after, [x_explosion, y_explosion])

# Plotting the trajectory before and after the explosion
plt.figure(figsize=(10, 6))

# Plot the trajectory before the explosion
plt.plot(x, y, label="Grenade Trajectory (before explosion)")

# Plot the trajectories of the two fragments
plt.plot(x_f1, y_f1, '--', label="Fragment 1 Trajectory (after explosion)")
plt.plot(x_f2, y_f2, '--', label="Fragment 2 Trajectory (after explosion)")

# Mark the explosion point
plt.scatter([x_explosion], [y_explosion], color='red', label="Explosion Point")

# Labels and title
plt.xlabel('x (m)')
plt.ylabel('y (m)')
plt.title('Grenade Trajectory with Explosion and Air Resistance')
plt.legend()
plt.grid(True)

# Display the plot
plt.show()

# Print velocities before the explosion and of each fragment
print(f"Velocity of grenade before explosion: {v_before_explosion:.2f} m/s")
print(f"Velocity of Fragment 1 after explosion: {np.linalg.norm(v_fragment1):.2f} m/s")
print(f"Velocity of Fragment 2 after explosion: {np.linalg.norm(v_fragment2):.2f} m/s")
